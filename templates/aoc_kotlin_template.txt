import aoc.handleSubmit
import extensions.printSeparated
import kotlin.time.measureTimedValue


fun main() {

     val config = ConfigForDay(
             submit1 = false,
             submit2 = false,
             check1 = false,
             check2 = false,
             checkDemo1 = false,
             checkDemo2 = false,
             execute1 = false,
             execute2 = false,
             execute1demo = true,
             execute2demo = false,
             exampleSolution1 = 0,
             exampleSolution2 = 0,
             solution1 = 0,
             solution2 = 0)

    fun part1(input: List<String>): Int {
        var result = 0
        return result
    }

    fun part2(input: List<String>): Int {
        var result = 0
        return result
    }

    try {
        val exampleInput1 = readInput("day_@@DAY_NUM_PADDED@@_demo", "@@AOC_YEAR@@")
        val exampleInput2 = readInput("day_@@DAY_NUM_PADDED@@_demo", "@@AOC_YEAR@@")

        if(config.execute1demo) {

            "Part 1 Demo".printSeparated()
